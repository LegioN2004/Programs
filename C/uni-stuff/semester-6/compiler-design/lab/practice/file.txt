//THIS IS A LEX PROGRAM TO PRACTICE
%{
#include <stdio.h>
#include <stdlib.h>
double a = 0.0, b = 0.0;
char op = '\0';
%}
digit   [0-9]
num     {digit}+(\.{digit}+)? 
%%
{num} {
         if (a == 0.0 && op == '\0')
             a = atof(yytext);
         else
             b = atof(yytext);
      }
[\+\-\*\/] { op = yytext[0]; }
[\n] {
         switch(op) {
             case '+': printf("Result: %lf\n", a + b); break;
             case '-': printf("Result: %lf\n", a - b); break;
             case '*': printf("Result: %lf\n", a * b); break;
             case '/': 
                         if(b != 0)
                             printf("Result: %lf\n", a / b);
                         else
                             printf("Division by zero error\n");
                         break;
             default:  printf("Unknown operator\n");
         }
         a = b = 0.0;
         op = '\0';
      }
.          { /* ignore */ }
%%
int main(void) {
    yylex();
    return 0;
}
